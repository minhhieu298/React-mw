const AssetReport = () => {
  const [activeIndex, setActiveIndex] = useState<number | null>(null);

  const handleMenuItemClick = (index: number) => {
    setActiveIndex(index);
  };

  const renderMenuItem = (item: MenuItem, index: number) => {
    const isActive = activeIndex === index;

    return (
      <li
        key={item.path}
        onClick={() => handleMenuItemClick(index)}
        // className={isActive ? 'active' : ''}
        className={`list-sub-menu ${isActive ? 'active' : '' }`}
      >
        <NavLink to={item.path}>{item.label}</NavLink>
        {item.children && (
          <ul className="sub-menu">
            {item.children.map((childItem, childIndex) => (
              <li key={childItem.path}>
                <NavLink to={childItem.path}>{childItem.label}</NavLink>
              </li>
            ))}
          </ul>
        )}
      </li>
    );
  };

  return (
    <nav>
      <ul>
        {menuItems.map((item, index) => renderMenuItem(item, index))}
      </ul>
    </nav>
  );
};

const DraggableDiv: React.FC = () => {
  const windowHeight = window.innerHeight - 40;
  const [heightW, setHeightW] = useState<number>(windowHeight);
  const [height, setHeight] = useState<number>(200);

  const draggingRef = useRef<boolean>(false);
  const yOffsetRef = useRef<number>(0);

  const handleDragStart = (e: React.DragEvent<HTMLDivElement>) => {
    draggingRef.current = true;
    yOffsetRef.current = e.clientY - height;
    e.dataTransfer?.setDragImage(new Image(), 0, 0); // ẩn hiệu ứng kéo thả mặc định
  };

  const handleDrag = (e: React.DragEvent<HTMLDivElement>) => {
    if (draggingRef.current) {
      const newHeight = e.clientY - yOffsetRef.current;
      requestAnimationFrame(() => {
        setHeight(newHeight);
      });
    }
  };

  const handleDragEnd = () => {
    draggingRef.current = false;
  };

  return (
    <div className="relative">
      <div
        style={{
          height: `${height}px`,
          backgroundColor: "grey",
     
        }}
      
      >
      div ngoài
      </div>
 
      <div
      className="red-div"
        style={{
          height: `${heightW - height}px`,
          backgroundColor: "red",
          cursor: "grab",
        }}
        onClick={(e) => e.stopPropagation()}
        draggable
        onDragStart={handleDragStart}
        onDrag={handleDrag}
        onDragEnd={handleDragEnd}
      >
          Drag me!
      </div>
    </div>
  );
};
